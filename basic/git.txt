# edit in txt file by " vi filename.txt " then for Save and quit/exit vi " :wq or ZZ " 
    --> " i " — Switch to Insert mode.
    --> " ESC " — Switch to Command mode.
    --> " ESC :w " — Save and continue editing.
    --> " ESC :wq or ZZ " — Save and quit/exit vi.
    --> " ESC :q! " — Quit vi and do not save changes.
    --> " ESC yy " — Yank (copy) a line of text.
    --> " ESC p " — Paste a line of yanked text below the current line.
    --> " ESC o " — Open a new line under the current line.
    --> " ESC O " — Open a new line above the current line.
    --> " ESC A " — Append to the end of the line.
    --> " ESC a " — Append after the cursor’s current position.
    --> " ESC I " — Insert text at the beginning of the current line.
    --> " ESC b " — Go to the beginning of the word.
    --> " ESC e " — Go to the end of the word.
    --> " ESC x " — Delete a single character.
    --> " ESC dd " — Delete an entire line.
    --> " ESC Xdd " — Delete X number of lines.
    --> " ESC Xyy " — Yank X number of lines.
    --> " ESC G " — Go to the last line in a file.
    --> " ESC XG " — Go to line X in a file.
    --> " ESC gg " — Go to the first line in a file.
    --> " ESC :num " — Display the current line’s line number.
    --> " ESC h " — Move left one character.
    --> " ESC j " — Move down one line.
    --> " ESC k " — Move up one line.
    --> " ESC l " — Move right one character.
